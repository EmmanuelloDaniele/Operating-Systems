#!/bin/bash

if[$# - le1];
then
	echo "pochi parametri"

elif[$# - ge3];
then	
	echo "troppi parametri"

else:
	if[-d $1];

		grep -n $2 *.txt					// prendo il numero della stringa (-n) e la stringa ($2) dal file txt
	else
		echo "directory non presente"
	fi
fi
